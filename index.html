const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const scoreElement = document.getElementById('score'); const modal = document.getElementById('math-modal'); const questionText = document.getElementById('question-text'); const answerInput = document.getElementById('answer-input'); canvas.width = 400; canvas.height = 600; let score = 0; let fruits = []; let gameActive = true; let currentAnswer = 0; const basket = { x: 160, y: 550, width: 80, height: 20, speed: 7 }; const keys = {}; // Kontrol Basket window.addEventListener('keydown', (e) => keys[e.code] = true); window.addEventListener('keyup', (e) => keys[e.code] = false); function spawnFruit() { if (!gameActive) return; const value = Math.floor(Math.random() * 10) + 1; fruits.push({ x: Math.random() * (canvas.width - 30), y: -30, radius: 20, value: value, speed: 2 + Math.random() * 2 }); } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Gambar Basket ctx.fillStyle = '#e67e22'; ctx.fillRect(basket.x, basket.y, basket.width, basket.height); // Update & Gambar Buah fruits.forEach((fruit, index) => { fruit.y += fruit.speed; ctx.beginPath(); ctx.arc(fruit.x, fruit.y, fruit.radius, 0, Math.PI * 2); ctx.fillStyle = '#e74c3c'; ctx.fill(); ctx.closePath(); ctx.fillStyle = 'white'; ctx.fillText(fruit.value, fruit.x - 5, fruit.y + 5); // Cek Tabrakan (Berhasil menangkap buah) if (fruit.y + fruit.radius > basket.y && fruit.x > basket.x && fruit.x < basket.x + basket.width) { score += fruit.value; scoreElement.innerText = score; fruits.splice(index, 1); // Cek jika skor kelipatan 10 untuk muncul soal if (score >= 10 && score % 10 < fruit.value) { triggerMathChallenge(); } } // Hapus jika jatuh ke bawah if (fruit.y > canvas.height) fruits.splice(index, 1); }); // Pergerakan Basket if (keys['ArrowLeft'] && basket.x > 0) basket.x -= basket.speed; if (keys['ArrowRight'] && basket.x < canvas.width - basket.width) basket.x += basket.speed; if (gameActive) requestAnimationFrame(draw); } function triggerMathChallenge() { gameActive = false; const num1 = Math.floor(Math.random() * 20); const num2 = Math.floor(Math.random() * 20); currentAnswer = num1 + num2; questionText.innerText = `Berapa ${num1} + ${num2}?`; modal.classList.remove('hidden'); } function checkAnswer() { const userAnswer = parseInt(answerInput.value); if (userAnswer === currentAnswer) { alert("Benar!"); } else { score -= 5; scoreElement.innerText = score; alert("Salah! Skor dikurangi 5"); } modal.classList.add('hidden'); answerInput.value = ''; gameActive = true; draw(); } setInterval(spawnFruit, 1500); draw();
